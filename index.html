Goroutine mit Wait Group


                        func HelloCongressman() {
                            fmt.Println("Hello Congressman", name)
                        }
                        
                        func main() {
                          var waitGroup sync.WaitGroup
                          waitGroup.Add(1)
                          go func() {
                              HelloCongressman("Russo")
                              waitGroup.Done()
                          }()
                          waitGroup.Wait()
                        }

Goroutine mit Wait Group und Defer


                        func HelloCongressman() {
                            fmt.Println("Hello Congressman", name)
                        }
                        
                        func main() {
                          var waitGroup sync.WaitGroup
                          waitGroup.Add(1)
                          go func() {
                              defer waitGroup.Done()
                              HelloCongressman("Russo")
                          }()
                          waitGroup.Wait()
                        }

Channel

                        
                        func main() {
                          money := make(chan int)
                          go Congressman(money)

                          // Nachricht senden
                          money <- 100
                        }

                        func Congressman(money chan int) {
                          // Nachricht empfangen
                          amount := <-money

                          fmt.Println("Received", amount, "$!")
                        }

Channel mit Select


                        func main() {
                          money := make(chan int)
                          go Congressman(money)

                          // Nachricht senden
                          money <- 100
                        }

                        func Congressman(money chan int) {
                          select {
                          case amount := <-money:
                            fmt.Println("Received", amount, "$!")
                          }
                        }

Channelmit Timeout


                        func main() {
                          money := make(chan int)
                          go Congressman(money)

                          time.Sleep(2 * time.Second)
                        }

                        func Congressman(money chan int) {
                          select {
                          case amount := <-money:
                            fmt.Println("Received", amount, "$!")
                          case <-time.After(1 * time.Second):
                            fmt.Println("Got nothing ...!")
                          }
                        }
