Channel

                        
                        func main() {
                          money := make(chan int)
                          go Congressman(money)

                          // Nachricht senden
                          money <- 100
                        }

                        func Congressman(money chan int) {
                          // Nachricht empfangen
                          amount := <-money

                          fmt.Println("Received", amount, "$!")
                        }

Channel mit Select


                        func main() {
                          money := make(chan int)
                          go Congressman(money)

                          // Nachricht senden
                          money <- 100
                        }

                        func Congressman(money chan int) {
                          select {
                          case amount := <-money:
                            fmt.Println("Received", amount, "$!")
                          }
                        }

Channelmit Timeout


                        func main() {
                          money := make(chan int)
                          go Congressman(money)

                          time.Sleep(2 * time.Second)
                        }

                        func Congressman(money chan int) {
                          select {
                          case amount := <-money:
                            fmt.Println("Received", amount, "$!")
                          case <-time.After(1 * time.Second):
                            fmt.Println("Got nothing ...!")
                          }
                        }
